create table users_test(
user_id serial primary key,
telegram_id varchar,
email varchar,
password varchar,
nick varchar,
nationality varchar,
premium boolean,
role varchar,
email_notifications boolean,
verified_email boolean,
telegram boolean,
website boolean
)

create table active_requests(
id serial primary key,
user_id varchar
)

create table awaited_messages(
id serial primary key,
user_id varchar,
request varchar
)

create table data_imports(
id serial primary key,
email varchar,
request varchar,
method varchar
)

create table email_verifications(
id serial primary key,
user_id int,
token varchar
)

create table link_requests(
id serial primary key,
request_id varchar,
link varchar,
expiration bigint
)

create table messages_history(
id serial primary key,
user_id int,
message varchar,
sent_date timestamp without time zone,
foreign key (user_id) references users_test (user_id)
)

create table notifications_chat(
id serial primary key,
to_user_id varchar,
message varchar,
sent_date timestamp without time zone
)

create table notifications_email(
id serial primary key,
to_email varchar,
message varchar,
sent_date timestamp without time zone
)

create table products_test(
product_id serial primary key,
product_name varchar,
description varchar,
price numeric,
variant varchar,
link varchar,
price_alert varchar,
user_id int,
currency varchar,
photo_url varchar,
service_name varchar,
notify_when_available boolean,
foreign key (user_id) references users_test (user_id)
)

create table updates_history(
id serial primary key,
product_id int,
price numeric,
last_update timestamp without time zone
)

create table waiting_users(
id serial primary key,
user_id int,
foreign key (user_id) references users_test (user_id)
)

create table waiting_variants(
id serial primary key,
url varchar
)

create table updated_variants(
id integer primary key,
url varchar,
variants varchar[],
error boolean,
error_message varchar
)

create table waiting_products(
id serial primary key,
url varchar,
variant varchar
)

create table updated_products(
id integer primary key,
product_name varchar,
description varchar,
photo_url varchar,
price numeric,
variant varchar,
link varchar,
price_alert varchar,
service_name varchar,
currency varchar,
error boolean,
error_message varchar
)

create table waiting_products_price(
id integer primary key,
new_price numeric,
link varchar,
variant varchar,
service_name varchar
)

create table updated_products_price(
id integer primary key,
new_price numeric
)
